    auto_ptr¶

    new иҜӯеҸҘеҲӣе»әзҡ„еҜ№иұЎпјҢеңЁ auto_ptr жҢҮй’Ҳй”ҖжҜҒзҡ„ж—¶еҖҷпјҢжҢҮй’ҲжҢҮеҗ‘зҡ„еҜ№иұЎд№ҹдјҡиў«й”ҖжҜҒ

    жүҖжңүжқғиҪ¬з§»пјҡ pa жҳҜдёҖдёӘ auto_ptr жҢҮй’ҲпјҢжҢҮеҗ‘еҜ№иұЎ objгҖӮзҺ°еңЁдёҖдёӘж–°зҡ„ auto_ptr жҢҮй’Ҳ pb жҢҮеҗ‘еҜ№иұЎ objпјҢеҲҷ pa жҢҮй’ҲдјҡеӨұж•ҲпјҲеҸҳжҲҗ nullptrпјү

    еҺҹзҗҶпјҡеңЁжӢ·иҙқпјҲиөӢеҖјпјүиҝҮзЁӢдёӯпјҢзӣҙжҺҘеүҘеӨәеҺҹеҜ№иұЎеҜ№еҶ…еӯҳзҡ„жҺ§еҲ¶жқғпјҢиҪ¬дәӨз»ҷж–°еҜ№иұЎпјҢз„¶еҗҺе°ҶеҺҹжҢҮй’ҲзҪ®дёә nullptr

    е®һйҷ…дёҠе°ұжҳҜжҠҠеҺҹжқҘзҡ„жҢҮй’ҲеҸҳжҲҗ NULLпјҢз„¶еҗҺж–°е»әдёҖдёӘжҢҮй’Ҳ

    int main() {
    auto_ptr<int> p(new int(10));
    cout << *p << endl;
    printf("------------------------------------------\n");

    auto_ptr<string> lang[5] = {
            auto_ptr<string>(new string("C")),
            auto_ptr<string>(new string("C++")),
            auto_ptr<string>(new string("Java")),
            auto_ptr<string>(new string("Rust")),
            auto_ptr<string>(new string("Python"))
    };

    for(int i=0; i<5; i++){
        cout << *lang[i] << endl;
    }
    printf("------------------------------------------\n");

    auto_ptr<string> pa;
    pa = lang[2];// еүҘеӨәжүҖжңүжқғ
    for(int i=0; i<5; i++){
        cout << *lang[i] << endl;
    }

    return 0;
}
